function partsSums($ls) {
  $count = count($ls);
  $i = 0;
  $sum = array_sum($ls);
  $result = [$sum];
  
  while ($count > 0) {
    $sum -= $ls[$i];
    $result[] = $sum;
    $count--;
    $i++;
  }
  if ($sum !== 0) {
    $result[] = 0;
  }
  return $result;
}
